대충 이런 형태로 SQL작성

    SELECT
        id,
        memo,
        updated_by,
        to_char(updated_at, 'YYYY/MM/DD HH24MIss') AS updated_at
    FROM
        memo
    WHERE
            memo_category = 'normal'
        AND logical_delete_flag = false
    ORDER BY
        id DESC
    ;
