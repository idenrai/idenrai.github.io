## 각 뉴스의 상세 내용 표시

### 상세 내용 가져오기

```python
for link in links:
    news_res = requests.get(link)
    news_soup = BeautifulSoup(news_res.text, "html.parser")

    news_elem = news_soup.find("p", class_="sc-kvjbNB")
    news_link = news_elem.a.attrs["href"]

    detail_res = requests.get(news_link)
    detail_soup = BeautifulSoup(detail_res.text, "html.parser")
#    detail_elem = detail_soup.find("p", class_="highLightSearchTarget")
    detail_text = detail_soup.find(class_=re.compile("yjSlinkDirectlink"))

    print(detail_soup.title.text)
    print(news_link)
#    print(detail_elem.text)
    print(detail_text.text if hasattr(detail_text, "text") else '', end="\n\n\n\n")
```
